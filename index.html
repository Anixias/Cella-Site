Cella
A minimalistic systems programming language
Token: type
{
	kind: TokenKind
	text: String
}

TokenKind: enum
{
	Identifier
	Keyword
	Literal
}

Lexer: type
{
	code: String
	
	scanTokens: fun(self): Token[]
	{
		var tokens = Token[]
		
		for (var i in 0..code.length)
		{
			val char = code[i]
			
			if with char
			{
				Char.isWhitespace
					cont
				Char.isLetter
					tokens += identifierToken(var i)
				Char.isDigit
					tokens += numberToken(var i)
			}
		}
	}
}

main: entry(args: String[]): Int32
{
	val lexer: Lexer("mod helloWorld\n\nuse std.io.Console\n\nmain: entry(args: String[]): Int32\n{\n\tConsole.println("Hello, World!")\n\tret 0}")
	for var token in lexer.scanTokens()
	{
		Console.println(token.text)
	}
}
